=========
>>> matrix = [[1, 2, 3], [4, 5, 6]]

# input of standard matrix and div = integer

# input of standard matrix and div = float
>>> print(matrix_divided(matrix, 1.5))
[[0.67, 1.33, 2.0], [2.67, 3.33, 4.0]]

>>> print(matrix)
